<script src="https://unpkg.com/leaflet/dist/leaflet.js"></script>
<script>
    // 1. Haritayı Mersin/Hatay merkezli başlat
    var map = L.map('map').setView([36.8, 34.6], 10); 
    L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png').addTo(map);

    // 2. Namaz Vakitlerini Getiren Fonksiyon (Örnek API kullanımı)
    function vakitleriGetir(sehir) {
        document.getElementById('sehir-adi').innerText = sehir.toUpperCase();
        // Buraya ezanvakti.herokuapp.com veya benzeri bir API bağlanacak
        document.getElementById('vakit-listesi').innerHTML = `
            <div class="vakit-kutu">İmsak: 05:42</div>
            <div class="vakit-kutu">Öğle: 12:55</div>
            <div class="vakit-kutu">İkindi: 15:38</div>
            <div class="vakit-kutu">Akşam: 18:12</div>
            <div class="vakit-kutu">Yatsı: 19:35</div>
        `;
    }

    // 3. Harita Hareket Ettikçe Şehri Algıla (Temsili)
    map.on('moveend', function() {
        // Haritanın ortasındaki koordinata göre şehir tahmini yapıp vakitleri güncelleyebilirsin
        vakitleriGetir("Mersin / Hatay Bölgesi");
    });

    vakitleriGetir("Mersin"); // Başlangıçta Mersin gelsin
</script>
